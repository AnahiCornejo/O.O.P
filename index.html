Paradigmas de la OOP en Python

Concepto Definición Ejemplo en Python
Encapsulamiento Ocultar los detalles internos de un objeto y exponer solo lo necesario. 
class Persona:
    def __init__(self, nombre, edad):
        self.__nombre = nombre  # Atributo privado
        self.__edad = edad      # Atributo privado

    def get_nombre(self):
        return self.__nombre

    def set_edad(self, edad):
        if edad > 0:
            self.__edad = edad

persona = Persona("Juan", 30)
print(persona.get_nombre())  # Imprime: Juan
persona.set_edad(31)
Abstracción Simplificar la complejidad al enfocarse en los atributos y comportamientos esenciales. 
from abc import ABC, abstractmethod

class Animal(ABC):
    @abstractmethod
    def hacer_sonido(self):
        pass

class Perro(Animal):
    def hacer_sonido(self):
        return "Ladrido"

perro = Perro()
print(perro.hacer_sonido())  # Imprime: Ladrido
Herencia Permitir que una clase herede propiedades y métodos de otra clase. 
class Vehiculo:
    def arrancar(self):
        return "El vehículo arranca"

class Coche(Vehiculo):
    def arrancar(self):
        return "El coche arranca"

coche = Coche()
print(coche.arrancar())  # Imprime: El coche arranca
Polimorfismo Permitir que un método se comporte de manera diferente según el objeto que lo invoque. 
class Figura:
    def dibujar(self):
        pass

class Circulo(Figura):
    def dibujar(self):
        return "Dibujando un círculo"

class Rectangulo(Figura):
    def dibujar(self):
        return "Dibujando un rectángulo"

def dibujar_figura(figura):
    print(figura.dibujar())

circulo = Circulo()
rectangulo = Rectangulo()

dibujar_figura(circulo)       # Imprime: Dibujando un círculo
dibujar_figura(rectangulo)    # Imprime: Dibujando un rectángulo
